select TenBenhNhan as HoVaTen
	, MaYTe
	,SoDienThoai
	,NamSinh
	,CMND_CCCD = ''
	,GioiTinh
	,DiaChi
    ,KetQua
	,KyThuatXetNghiem
	,LyDoXetNghiem
	,LoaiMau
	,ThoiGianLayMau
	,ThoiGianKetQua
	,MaXetNghiem
	,MaCskcb
	,PcrValue
	,IDRef
    ,ThanhToan
from
(
	select distinct
		bn.TenBenhNhan				
		,bn.MaYTe
		,bn.SoDienThoai
		,NamSinh = bn.NamSinh
		,GioiTinh  =  case when bn.GioiTinh = 'T' then 0 else 1 end		
		,bn.DiaChi
        ,KetQua = case when sl.KetQua = '0' then '2' else '3' end
		,KyThuatXetNghiem = 1
		,LyDoXetNghiem = 1
		,LoaiMau = 0
		--,DoiTuong = dt.TenDoiTuong
		--,DichVu = dv.TenDichVu
		,ThoiGianLayMau = replace(replace(replace(convert(varchar(16),sl.tglaymau,120),'-',''),':',''),' ','')
        ,ThoiGianKetQua = replace(replace(replace(convert(varchar(16),sl.TgIn,120),'-',''),':',''),' ','')
		,MaXetNghiem = ''
		,MaCskcb = '79422'
		,PcrValue = ''
        --,ThoiGianIn = sl.tgin
        ,Id = sl.Id
		,sl.IDRef
        --,CodeXn = sl.CodeXn
        --,NguoiLayMau = nv.TenNhanVien
		,ThanhToan = case when hd.HoaDon_Id is not null and hd.HuyHoaDon = 0 and hd.HoanTra = 0 then 0 else 1 end

	from CLSYeuCau yc
		left join CLSYeuCauChiTiet ct on ct.CLSYeuCau_Id = yc.CLSYeuCau_Id
		left join DM_DichVu dv on dv.DichVu_Id = ct.DichVu_Id
left join NhanVien nv on nv.NhanVien_Id = yc.BacSiChiDinh_Id
		left join TiepNhan tn on tn.TiepNhan_Id = yc.TiepNhan_Id
		left join DM_DoiTuong dt on dt.DoiTuong_Id = tn.DoiTuong_Id
		left join DM_BenhNhan bn on bn.BenhNhan_Id = yc.BenhNhan_Id	
left join HoaDonChiTiet hdct on hdct.YeuCauChiTiet_Id = ct.YeuCauChiTiet_Id
		left join HoaDon hd on hd.HoaDon_Id = hdct.HoaDon_Id
        inner join SoLanIn sl on sl.IDRef = yc.CLSYeuCau_Id
	where yc.ThoiGianYeuCau between '{0}' and '{1}'
		and ct.DichVu_Id in (803,1375,1410,1413)
        and yc.CLSYeuCau_Id in (select IDRef from SoLanIn where IDRef = yc.CLSYeuCau_Id)
        and sl.LayMau = 1
        and sl.ketqua is not null
        and sl.DaIn is not null and sl.tgin is not null
) chitiet		
		group by TenBenhNhan
	, MaYTe
	,SoDienThoai
	,NamSinh
	,GioiTinh
	,DiaChi
    ,KetQua
	,KyThuatXetNghiem
	,LyDoXetNghiem
	,LoaiMau
	,ThoiGianLayMau
	,ThoiGianKetQua
	,MaXetNghiem
	,MaCskcb
	,PcrValue
	,IDRef
    ,ThanhToan
	order by ThoiGianKetQua
